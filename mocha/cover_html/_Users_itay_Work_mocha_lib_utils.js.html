  /**
   * Escape special characters in the given string of html.
   *
   * @param  {String} html
   * @return {String}
   * @api private
   */
  
  exports.escape = function(html) {
    return String(html)
      .replace(/&/g, '&')
      .replace(/"/g, '"')
      .replace(/</g, '<')
      .replace(/>/g, '>');
